CREATE TABLE administrador(
	cedula integer,
	nombre varchar(30),
	profesion varchar(40),
	salario numeric(10,2),
	PRIMARY KEY(cedula)
);
CREATE TABLE hotel(
	nit integer,
	nombre varchar(30) not null,
	direccion varchar(40),
	telefono varchar(15),
	numero_habitaciones int,
	PRIMARY KEY(nit)
);
CREATE TABLE huesped(
	cedula integer,
	nombre varchar(30),
	PRIMARY KEY(cedula)
);
CREATE TABLE reserva(
	codigo integer,
	fecha_ingreso date,
	hora_ingreso time,
	numero_dias integer,
	total_reserva numeric(10,2),
	cedula_huesped integer,
	cedula_administrador integer,
	nit_hotel integer,
	PRIMARY KEY(codigo),
	FOREIGN key(cedula_huesped) REFERENCES huesped,
	FOREIGN key(cedula_administrador) REFERENCES administrador,
	FOREIGN key(nit_hotel) REFERENCES hotel
);
